\section{Scope}

\paragraph{Target Lean declarations.}

\section{Ambient assumptions}

Fix a finite group $G$ and a field $k$ whose characteristic does not divide the
order of $G$.

\section{Declaration blueprint}

\begin{enumerate}
\item \texttt{exists\_simple\_summand}: every nonzero finite-dimensional
representation contains a simple summand.
\item \texttt{representation\_directSum\_of\_simple}: semisimple decomposition.
\item \texttt{character\_decomposition}: expansion against irreducible
characters.
\item \texttt{fdRep\_equiv\_groupAlgebraMod}: bridge to modules over $k[G]$.
\end{enumerate}

\section{Seed theorem cluster}

\begin{lemma}[Simple summand]\label{lem:simple-summand}
Every nonzero finite-dimensional representation of $G$ over $k$ contains a
simple subrepresentation that splits off as a direct summand.
\end{lemma}

\begin{theorem}[Semisimple decomposition]
\label{thm:semisimple-decomposition}
Every finite-dimensional representation of $G$ over $k$ is a finite direct sum
of simple representations.
\end{theorem}

\begin{corollary}[Character decomposition]\label{cor:character-decomposition}
The character of a finite-dimensional representation decomposes as a finite
linear combination of irreducible characters.
\end{corollary}

\section{Formalization order}

\begin{enumerate}
\item package the simple-summand statement
\item build the finite direct-sum decomposition
\item derive the character decomposition
\item add the bridge to group-algebra modules
\end{enumerate}
