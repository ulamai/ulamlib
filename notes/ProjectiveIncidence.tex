\section{Scope}

This note targets an incidence layer over existing projectivization objects.
The intended Lean destination is
\texttt{ULAM/Geometry/ProjectiveIncidence.lean}.

\paragraph{Target Lean declarations.}
This note should eventually feed declarations with names close to
\texttt{Collinear}, \texttt{lineThrough}, \texttt{lineThrough\_unique}, and
\texttt{ProjectiveMap.preservesCollinear}.

\section{Ambient assumptions}

Fix a field $K$ and a nontrivial finite-dimensional vector space $V$ over $K$.
Projective space means the projectivization of $V$.

\section{Declaration blueprint}

\begin{enumerate}
\item \texttt{Collinear}: ternary collinearity relation on projective points.
\item \texttt{lineThrough}: projective line generated by two points.
\item \texttt{memLineThroughLeft}: the left point lies on its generated line.
\item \texttt{memLineThroughRight}: the right point lies on its generated line.
\item \texttt{lineThrough\_unique}: uniqueness of the line through two distinct
points.
\item \texttt{ProjectiveMap.preservesCollinear}: incidence-preservation API.
\end{enumerate}

\section{Core definitions}

\begin{definition}[Collinearity]\label{def:collinearity}
Three points in projective space are collinear if they lie on a common
projective line.
\end{definition}

\begin{definition}[Line through two points]\label{def:line-through-two-points}
Given two distinct points in projective space, the line through them is the
projectivization of the two-dimensional linear span of representatives of those
points.
\end{definition}

\section{Seed theorem cluster}

\begin{lemma}[Basic properties of collinearity]
\label{lem:basic-properties-of-collinearity}
Collinearity is symmetric in its arguments and every triple containing repeated
points is collinear.
\end{lemma}

\begin{lemma}[Endpoints lie on the generated line]
\label{lem:endpoints-lie-on-generated-line}
Each of the two generating points lies on the line determined by those points.
\end{lemma}

\begin{theorem}[Two points determine a line]\label{thm:two-points-determine-a-line}
Two distinct points of projective space lie on a unique projective line.
\end{theorem}

\begin{theorem}[Projectivized linear maps preserve incidence]
\label{thm:projectivized-linear-maps-preserve-incidence}
Projectivizations of injective linear maps preserve collinearity and incidence.
\end{theorem}

\begin{corollary}[Projective-space axioms]\label{cor:projective-space-axioms}
The standard projective space attached to a vector space satisfies the basic
incidence axioms of projective geometry.
\end{corollary}

\section{Formalization order}

\begin{enumerate}

\item Prove local incidence lemmas for generated lines.
\item Prove uniqueness of the line through two distinct points.

\end{enumerate}
