\section{Ambient assumptions}

\section{Core definitions}

\begin{definition}[$C_0(X,\mathbb{R})$]\label{def:c0-space}
Let $X$ be a locally compact Hausdorff space. The algebra $C_0(X, \mathbb{R})$
consists of continuous real-valued functions on $X$ that vanish at infinity.
\end{definition}

\begin{definition}[Separates points]\label{def:separates-points}
A subalgebra $A \subseteq C_0(X, \mathbb{R})$ separates points if for every
distinct pair $x,y \in X$ there exists $f \in A$ such that $f(x) \neq f(y)$.
\end{definition}

\begin{definition}[Vanishes nowhere]\label{def:vanishes-nowhere}
A subset $A \subseteq C_0(X, \mathbb{R})$ vanishes nowhere if for every
$x \in X$ there exists $f \in A$ such that $f(x) \neq 0$.
\end{definition}

\section{Seed theorem cluster}

\begin{lemma}[Absolute value implies lattice closure]
\label{lem:absolute-value-implies-lattice-closure}
If a real subalgebra of $C_0(X, \mathbb{R})$ is closed under absolute value,
then it is closed under pointwise supremum and infimum.
\end{lemma}

\begin{lemma}[Closed-sublattice packaging]\label{lem:closed-sublattice-packaging}
A uniformly closed real subalgebra of $C_0(X,\mathbb{R})$ that is closed under
absolute value determines a closed sublattice of $C_0(X,\mathbb{R})$.
\end{lemma}

\begin{theorem}[Real $C_0$ Stone-Weierstrass]\label{thm:real-c0-stone-weierstrass}
Let $X$ be locally compact Hausdorff and let
$A \subseteq C_0(X, \mathbb{R})$ be a subalgebra that separates points and
vanishes nowhere. Then $A$ is dense in $C_0(X, \mathbb{R})$ for the supremum
norm.
\end{theorem}

\begin{corollary}[Approximation corollary]\label{cor:compact-support-approximation}
Compactly supported continuous functions can be uniformly approximated by
elements of any dense subalgebra satisfying the hypotheses above.
\end{corollary}
